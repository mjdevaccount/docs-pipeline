weasyprint>=60.0  # Default PDF renderer (fast, but limited SVG foreignObject support)
playwright>=1.40.0  # FAANG-grade PDF renderer (perfect SVG rendering including foreignObject)

colorama>=0.4.6  # Optional: colored terminal output
tqdm>=4.65.0  # Optional: progress bars for batch processing

# Note: System dependencies required (see SETUP.md):

#   Linux: sudo apt install pandoc
#
# - Mermaid-CLI (https://github.com/mermaid-js/mermaid-cli)
#   npm install -g @mermaid-js/mermaid-cli
#
# - GTK Runtime (Windows only, for WeasyPrint)
#   Option 1: Install MSYS2, then: pacman -S mingw-w64-x86_64-gtk3
#   Option 2: Use Chocolatey: choco install gtk-runtime
